def ngram(x, n):
    return list(zip(*[x[i:] for i in range(n)]))

def build(X):
    model = {}
    for x in X:
        for i, k in enumerate(ngram(x, 2)):
            model.setdefault(i, {})
            model[i].setdefault(k, 0)
            model[i][k] += 1
    return {
        i: {k: v / max(x.values()) for k, v in x.items()}
        for i, x in model.items()
    }

model = build(names)

def check(model, x):
    p = [
        model.get(i, {}).get(k, 0)
        for i, k in enumerate(ngram(x, 2))
    ]
    return sum(p) / len(p)

check(model, 'kamuolys')
