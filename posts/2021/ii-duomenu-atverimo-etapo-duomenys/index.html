def fix_filename_encoding(p: pathlib.Path) -> pathlib.Path:
    return pathlib.Path(
        bytes(p).
        decode('ISO-8859-1').
        replace('+¦', 'ų').
        replace('-ù', 'ė').
        replace('+¥', 'ž').
        replace('+í', 'š').
        replace('-Ö', 'ę').
        replace('-à', 'ą').
        replace('-»', 'į').
        replace('-«', 'Į')
    )

docs = [fix_filename_encoding(p) for p in path.glob('**/*')]

docs = pd.DataFrame({
    'path': [str(p) for p in docs],
    'ext': [p.suffix for p in docs],
    'name': [p.name for p in docs],
})
